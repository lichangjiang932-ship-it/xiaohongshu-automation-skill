## 快速生成函数

```python
def 发布小红书纯文本(标题, 正文, 标签=None):
    """
    快速发布纯文本到小红书
    
    参数:
      标题: 字符串，≤20字
      正文: 字符串，≤1000字
      标签: 列表，8-12个标签
    
    返回:
      发布结果字典
    """
    
    # 检查字数
    if len(标题) > 20:
        return {"错误": f"标题超限：{len(标题)}/20字"}
    
    if len(正文) > 1000:
        return {"错误": f"正文超限：{len(正文)}/1000字"}
    
    # 默认标签
    if 标签 is None:
        标签 = ["知识分享", "经验总结", "实用技巧"]
    
    # 构建内容
    内容 = f"{标题}\n\n{正文}\n\n#{' #'.join(标签)}"
    
    return {
        "成功": True,
        "标题": 标题,
        "正文字数": len(正文),
        "标签数": len(标签),
        "内容预览": 内容[:200] + "..."
    }
```

def 生成排行榜(主题, 年份, 数据):

    正文 += "- 初学者：推荐1\n"
    正文 += "- 专业者：推荐2\n"
    
    正文 += "\n👇 你在用哪些工具？\n评论区分享！"
    
    return 标题, 正文

## 质量控制

### 快速检查
```python
def 检查内容(标题, 正文):
    """快速检查内容质量"""
    
    检查结果 = {
        "标题长度": len(标题) <= 20,
        "正文字数": len(正文) <= 1000,
        "有表情符号": any(c in 正文 for c in ["🚀", "🏆", "📊", "🎯"]),
        "有互动引导": "评论" in 正文 or "分享" in 正文,
        "有标签位置": "#" in 正文 or "标签" in 正文
    }
    
    通过 = all(检查结果.values())
    详情 = {项: "✅" if 结果 else "❌" for 项, 结果 in 检查结果.items()}
    
    return 通过, 详情
```

### 自动优化
```python
def 优化内容(内容):
    """自动优化内容格式"""
    
    # 确保有表情符号开头
    if not 内容.startswith("🚀"):
        内容 = "🚀 " + 内容
    
    # 确保有互动引导
    if "评论" not in 内容 and "分享" not in 内容:
        内容 += "\n\n👇 你有什么看法？评论区聊聊！"
    
    # 确保有标签
    if "#" not in 内容:
        内容 += "\n\n#知识分享 #经验总结"
    
    return 内容
```

### 重试机制
```python
def 带重试的发布(标题, 正文, 最大重试=3):
    """带重试的发布函数"""
    
    for 尝试次数 in range(最大重试):
        try:
            结果 = 发布小红书纯文本(标题, 正文)
            if 结果["成功"]:
                return 结果
        except Exception as 错误:
            print(f"第{尝试次数+1}次尝试失败：{错误}")
    
    return {"成功": False, "消息": "发布失败，已保存为草稿"}
```
